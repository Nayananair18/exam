pipeline {
  agent any 
  environment {
    PROJECT_DIR = "/home/student/hello-world" 
  }      
  stages {
    stage('Build') {
      steps {                 
        dir("${PROJECT_DIR}") {                     
          sh '''                     
          export PATH=$HOME/maven/bin:$PATH                     
          mvn clean package                     
          '''                
        }             
      }        
    }          
    stage('Verify') {
      steps {                 
        sh 'ls -l /home/student/hello-world/target/'             
      }         
    }          
    stage('Deploy') {             
      steps {                 
        sh '''                 
        export LC_ALL=C.UTF-8                 
        export LANG=C.UTF-8                 
        export PATH=$HOME/.local/bin:$PATH                  
        ansible-playbook -i /home/student/hosts.ini /home/student/setup.yml                 
        '''             
      }         
    }     
}
